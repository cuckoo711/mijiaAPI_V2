### 2. 自动刷新凭据

```python
from mijiaAPI_V2 import create_auth_service, create_api_client
from mijiaAPI_V2.domain.exceptions import TokenExpiredError

def get_api_client():
    """获取API客户端，自动处理凭据过期"""
    auth_service = create_auth_service()
    credential = auth_service.load_credential()
    
    # 检查凭据是否过期
    if credential.is_expired():
        print("凭据已过期，正在刷新...")
        credential = auth_service.refresh_credential(credential)
        auth_service.save_credential(credential)
    
    return create_api_client(credential)

# 使用
api = get_api_client()
```

### 3. 多用户凭据管理

```python
from typing import Dict
from mijiaAPI_V2.domain.models import Credential

class CredentialManager:
    """凭据管理器"""
    
    def __init__(self, base_dir: Path):
        self.base_dir = base_dir
        self.store = FileCredentialStore()
    
    def save_user_credential(self, user_id: str, credential: Credential):
        """保存用户凭据"""
        path = self.base_dir / f"{user_id}_credential.json"
        self.store.save(credential, str(path))
    
    def load_user_credential(self, user_id: str) -> Optional[Credential]:
        """加载用户凭据"""
        path = self.base_dir / f"{user_id}_credential.json"
        return self.store.load(str(path))
    
    def load_all_credentials(self) -> Dict[str, Credential]:
        """加载所有用户凭据"""
        credentials = {}
        for file in self.base_dir.glob("*_credential.json"):
            user_id = file.stem.replace("_credential", "")
            credential = self.store.load(str(file))
            if credential:
                credentials[user_id] = credential
        return credentials

# 使用
manager = CredentialManager(Path("~/.mijia/users"))
manager.save_user_credential("user_a", credential_a)
credentials = manager.load_all_credentials()
```

## 错误处理

### 1. 分层错误处理

```python
from mijiaAPI_V2.domain.exceptions import (
    TokenExpiredError,
    DeviceNotFoundError,
    NetworkError,
    MijiaAPIException
)

def control_device_safely(api, device_id, siid, piid, value):
    """安全的设备控制"""
    try:
        return api.control_device(device_id, siid, piid, value)
    
    except TokenExpiredError:
        # 凭据过期：刷新凭据后重试
        print("凭据过期，正在刷新...")
        refresh_credential()
        return api.control_device(device_id, siid, piid, value)
    
    except DeviceNotFoundError:
        # 设备不存在：记录错误，返回失败
        logger.error(f"设备不存在: {device_id}")
        return False
    
    except NetworkError as e:
        # 网络错误：记录错误，可能重试
        logger.error(f"网络错误: {e}")
        return False
    
    except MijiaAPIException as e:
        # 其他API错误：记录详细信息
        logger.error(f"API错误: {e.message}", exc_info=True)
        return False
```

### 3. 优雅降级

```python
def get_devices_with_fallback(api, home_id):
    """获取设备，失败时返回空列表"""
    try:
        return api.get_devices(home_id=home_id)
    except MijiaAPIException as e:

def get_device_statistics_with_cache(api, home_id):
    """获取统计信息，失败时使用缓存"""
    cache_key = f"stats:{home_id}"
    
    try:
        stats = api.get_device_statistics(home_id)
        cache.set(cache_key, stats, ttl=3600)
        return stats
    except MijiaAPIException as e:
        logger.warning(f"获取统计失败，尝试使用缓存: {e}")
        cached = cache.get(cache_key)
        if cached:
            return cached
        # 返回默认值
        return {"total": 0, "online": 0, "offline": 0}
```

## 性能优化

### 1. 使用缓存

```python
# ✅ 好：利用缓存
def get_all_devices(api, home_ids):
    """获取多个家庭的设备（利用缓存）"""
    all_devices = []
    for home_id in home_ids:
        # 第一次从API获取，后续从缓存获取
        devices = api.get_devices(home_id)
        all_devices.extend(devices)
    return all_devices

# ❌ 不好：每次都清空缓存
def get_all_devices_no_cache(api, home_ids):
    all_devices = []
    for home_id in home_ids:
        cache.clear()  # 不要这样做！
        devices = api.get_devices(home_id)
        all_devices.extend(devices)
    return all_devices
```

```python
import asyncio

# ✅ 好：并发获取多个家庭的设备
async def get_all_devices_async(api, home_ids):
    """异步并发获取设备"""
    tasks = [api.get_devices(home_id) for home_id in home_ids]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # 过滤错误
    devices = []
    for result in results:
        if isinstance(result, Exception):
            logger.error(f"获取设备失败: {result}")
        else:
            devices.extend(result)
    
    return devices

# ❌ 不好：顺序执行
async def get_all_devices_slow(api, home_ids):
    devices = []
    for home_id in home_ids:
        result = await api.get_devices(home_id)
        devices.extend(result)
    return devices
```

```python
def log_performance(operation: str):
```

```python
def validate_production_config(config: ConfigManager):
    """验证生产环境配置"""
    # 检查Redis配置
```

```python
import time
from dataclasses import dataclass
from typing import List

@dataclass
class Metric:
    name: str
    value: float
    timestamp: float

class MetricsCollector:
    """指标收集器"""

        self.metrics: List[Metric] = []

        """记录指标"""

            value=value,
```
